53 typedef struct {

55  struct {

57  uint16_t MASK:1; // bit 7
58  uint16_t OPCODE:4; // bits 8.. 11

64  uint16_t ZERO:8; // bits 8.. 15
65  } pos;
66  } param; // the initial parameters of the header
67  uint64_t length; // actual message length
68  union {
69  char part[4]; // the mask, array
70  uint32_t full; // the mask, all 32 bits
71  } key; // masking key
72  bool received; // was a message successfully received?
73 } ws_header_t;
